% --- Identity ---
\section{Identity}

\begin{frame}{Brand identity}
  \CollinsEyebrow{Who we are}

  \vspace{0.8em}
  \displayheading{Collins}\\
  \CollinsSubtitle{Boost your digital marketing and automation for faster business growth.}

  \vspace{1.2em}
  \begin{collinsblock}{Category}
    Senior-led digital marketing agency for growth brands: strategy, content, paid media, automation, and analytics as one accountable team.
  \end{collinsblock}

  \vspace{0.6em}
  \CollinsAccent{Studio polish, solo speed} for brands that outgrew freelancers but refuse enterprise bloat.
\end{frame}

\begin{frame}{Mission and positioning}
  \CollinsEyebrow{Mission}

  Collins helps founders and marketing leads unify SEO, content, paid media, and automation without global agency overhead or in-house ramp time. We replace fragmented freelancers with transparent retainers, senior strategists on every call, and one growth plan.

  \vspace{0.8em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{itemize}
      \item One team, full funnel
      \item Senior access by default
      \item Transparent partnerships
    \end{itemize}
    \column{0.48\textwidth}
    \begin{itemize}
      \item Speed without theater
      \item Neon clarity over noise
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Audience and values}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \CollinsEyebrow{Primary audience}
    \vspace{0.4em}

    Founders and marketing leads at B2B and DTC growth brands (\$1M--\$50M revenue) with active paid and content programs ready to unify.

    \vspace{0.8em}
    \CollinsEyebrow{Not for}
    \vspace{0.4em}

    {\color{collinsSecondary}Enterprise procurement cycles, cheapest hourly labor, or teams seeking a self-serve software platform.}

    \column{0.46\textwidth}
    \CollinsEyebrow{Core values}
    \vspace{0.4em}
    \begin{enumerate}
      \item Clarity
      \item Accountability
      \item Senior craft
      \item Speed
      \item Partnership
    \end{enumerate}
  \end{columns}
\end{frame}

\begin{frame}{Archetype and narrative}
  \CollinsEyebrow{Sage (85\%) / Creator (15\%)}

  Collins leads with calm authority and evidence-based guidance. Creator shows up in editorial systems, campaign frameworks, and demo experiences that feel crafted, not templated.

  \vspace{0.8em}
  \begin{collinsblock}{Transformation story}
    \textbf{Before:} Fragmented vendors, manual reporting, rising CAC.\\
    \textbf{After:} One senior-led team, shared playbook, cleaner pipeline.\\
    \textbf{Collins role:} Embedded growth partner without enterprise overhead.
  \end{collinsblock}
\end{frame}

% --- Voice ---
\section{Voice}

\begin{frame}{Voice attributes}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{itemize}
      \item \textbf{Direct} Say what we do and what it costs.
      \item \textbf{Operator-led} Practitioners, not pitch decks.
      \item \textbf{Evidence-based} Claims tie to outcomes.
    \end{itemize}
    \column{0.48\textwidth}
    \begin{itemize}
      \item \textbf{Warm but efficient} Approachable without filler.
      \item \textbf{Anti-bloat} Call out complexity and vanity metrics.
    \end{itemize}
  \end{columns}

  \vspace{1em}
  \CollinsEyebrow{Example}
  \vspace{0.3em}

  {\color{collinsSecondary}Don't:} ``Flexible engagement models.''\\
  {\color{collinsGreen}Do:} ``Retainers are quoted upfront with line-item clarity.''
\end{frame}

\begin{frame}{Tone and vocabulary}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \CollinsEyebrow{Tone by context}
    \vspace{0.4em}
    \begin{itemize}
      \item \textbf{Landing:} Confident, one promise per section
      \item \textbf{Proposals:} Plain language, line items
      \item \textbf{Demo captions:} Short beats, forward motion
      \item \textbf{Email:} Helpful, fast reply energy
    \end{itemize}
    \column{0.48\textwidth}
    \CollinsEyebrow{Use}
    \vspace{0.4em}
    {\small strategy call, retainer, sprint, full-funnel, senior-led, transparent scope, pipeline, CAC, playbook, ship, clarity}

    \vspace{0.6em}
    \CollinsEyebrow{Avoid}
    \vspace{0.4em}
    {\small\color{collinsSecondary} platform, free trial, sign up, dashboard login, synergy, cutting-edge orchestration}
  \end{columns}
\end{frame}

% --- Motion ---
\section{Motion}

\begin{frame}{Motion principles}
  \CollinsEyebrow{Restrained neon motion}

  \vspace{0.6em}
  \begin{itemize}
    \item Motion confirms intent. It never competes with copy or CTAs.
    \item Glow pulses only on demo stage and active carousel dots.
    \item Primary buttons: hover lift and glow only. No infinite loops.
    \item Hyperframe beats crossfade posters and stagger subtitle captions.
    \item Scroll reveals and stat count-ups run once on load.
    \item Honor \texttt{prefers-reduced-motion} with instant states.
  \end{itemize}

  \vspace{0.6em}
  \begin{collinsblock}{Beat timing}
    Crossfade duration: 420ms. Subtitle stagger: 45ms steps. Full spec in \texttt{docs/design.md}.
  \end{collinsblock}
\end{frame}

% --- Do's and Don'ts ---
\section{Guidelines}

\begin{frame}{Do's and don'ts: copy}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \CollinsEyebrow{Do}
    \vspace{0.4em}
    \begin{itemize}
      \item One concrete claim and one proof point per section
      \item Use ``Book a strategy call'' as primary CTA
      \item Name the six canonical services
      \item Quote transparent pricing philosophy
      \item Attribute testimonials to role and company
    \end{itemize}

    \column{0.48\textwidth}
    \CollinsEyebrow{Don't}
    \vspace{0.4em}
    \begin{itemize}
      \item Stack abstract superlatives without evidence
      \item Use product signup language
      \item Invent software features Collins does not sell
      \item Imply opaque contact-sales pricing
      \item Use anonymous customer filler
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Do's and don'ts: visual}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \CollinsEyebrow{Do}
    \vspace{0.4em}
    \begin{itemize}
      \item Apply neon tokens for CTAs and eyebrows
      \item Plus Jakarta Sans + Inter only
      \item Keep dark backgrounds and subtle green borders
      \item Preserve logo clear space and SVG glow
      \item Respect reduced-motion preferences
    \end{itemize}

    \column{0.48\textwidth}
    \CollinsEyebrow{Don't}
    \vspace{0.4em}
    \begin{itemize}
      \item Revert to legacy blue or slate palettes
      \item Introduce decorative serif fonts
      \item Ship light-mode SaaS layouts on the site
      \item Flatten the mark onto low-contrast imagery
      \item Hardcode hex values outside CSS tokens
    \end{itemize}
  \end{columns}
\end{frame}
